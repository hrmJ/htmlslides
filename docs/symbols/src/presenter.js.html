 41     /** 
 42      * Processes the call from the user to move to the next or to the previous
 43      * slide. 
 44      *
 45      * The function starts from the last item of the content chain (marked by
 46      * the {@link GetContentChain} function).  "The last item of the chain"
 47      * here means the lowest element in the current hierarchy. For example:
 48      *
 49      * The user is currently viewing a song in the "worship songs" -section. 
 50      *
 51      * That is, the chain is an array of this form:
 52      * [{@link MajakkaMessu}, {@link Section}, {@link SectionItem}, {@link SongContent}]
 53      *
 54      * 1. The Song is an iterable content, that iterates throug verses
 55      * 2. The SectionItem is also iterable, it's the container of the song, consisting of the Songs's title slide and the song
 56      * 3. The Section is the "ylistyslaulut" section of the service, also iterable (consists of multiple SectionItems)
 57      * 4. Finally, the MajakkaMessu item is also iterable, consisting of multiple sections
 58      *
 59      * So, when moving, the program needs to first move the lowest element in
 60      * the direction specified by the user. If the direction is "increment" and we come to the 
 61      * last verse and after that try to move forward, the function will recognize, that
 62      * instead of moving the lowest element in the hierarchy, it now has to move the next item.
 63      * Now, when the song has ended, the Mover tries to increment the section item instead.
 64      * Since the song was also the last element of the section item (consisting
 65      * of the Song's header and the song), the Mover continues up in the hierarchy and tries to
 66      * move the pointer in the Section. if this isn't the last song, the pointer will now, finally,
 67      * move forward.
 68      *
 69      *
 70      * @param {string} movetype - either "decrement" or "increment" 
 71      * 
 72      */
 73     this.Move = function(movetype){
 74             //Make sure the presentation is marked as started
 75             this.pointer.started = true;
 76             //increment pointers
 77             chain_idx = this.chain.length - 1;
 78             while(chain_idx >= 0){
 79                 thisobject = this.chain[chain_idx];
 80                 if(['decremented','incremented','started'].indexOf(thisobject.pointer.Move(movetype)) > -1){
 81                     break;
 82                 }
 83                 chain_idx--;
 84             }
 85 
 86             while(typeof thisobject.Show === 'undefined'){
 87                 //Iterating down to first showable content
 88                 thisobject = thisobject.current;
 89             }
 90             thisobject.Show();
 91             this.GetContentChain();
 92     }

 95     /** 
 96      * Iterates through the structure of the presentation from top to bottom.
 97      * Starts with the actual instance of the presentation class, then
 98      * moves down to the subitem currently active in the parent item
 99      * until there no longer are any subitems. A typical chain looks like this:
100      *
101      * 1. Presentation
102      * 2. Section (Introduction, Worship and prayer...)
103      * 3. Song
104      * 4. Verse
105      *
106      */
107     this.GetContentChain = function(){
108             //
109             //
110             thisobject = this;
111             this.chain = [thisobject];
112             try{
113                 while (thisobject.hasOwnProperty('current')){

115                     this.chain.push(thisobject);
116                 }
117             }

447 /**
448  * Pointers keep track of a set of contents in order to show them on the screen 
449  * on the right moment. Classes {@link Presentation} and {@link ScreenContent}
450  * have pointers. Basically, the simplest way to use a pointer is to move one slide
451  * forward or backward in the presentation. The usefulness of pointers is best demonstrated,
452  * when the presentation is a structured one, i.e. has nested contents. Having pointers as 
453  * a separate class is what makes the presentation non-linear, i.e. makes jumping and linkking 
454  * possible.
455  *
456  * @constructor
457  *
458  **/
459 function Pointer(pointed){
460     this.max  = pointed.items.length;
461     this.pointed = pointed;
462     this.started = false;
463     this.position = 0;
464     /**
465      *
466      * Moves the pointer forward or backward, if possible.
467      *
468      * @param {string} movetype - "increment", "decrement" or "unchanged"
469      *
470      **/
471     this.Move = function(movetype){
472         var returnvalue = false;
473         //Make sure the value is interpreted corretly after jumping from links etc
474         this.position = parseInt(this.position);
475         switch(movetype){
476             case "increment":
477                 if(this.position +1 < this.max){
478                     this.position++;
479                     this.started = true;
480                     returnvalue =  "incremented";
481                 }
482                 else{
483                     if(!this.started){
484                         this.started = true;
485                         if (typeof this.pointed.Show === 'undefined'){
486                             //if only one content object and this object not showable
487                             returnvalue =  false;
488                         }
489                         else{
490                             returnvalue =  "started";
491                         }
492                     }
493                 }
494                 break;
495             case "decrement":
496                 if(this.position -1 >= 0){
497                     this.position--;
498                     returnvalue = "decremented";
499                 }
500                 else{
501                     returnvalue = false;
502                 }
503                 break;
504             case "unchanged":
505                 returnvalue = "started";
506                 break;
507             default:
508                 break;
509         }
510 
511         //Set the parent object's currently active element
512         if (this.pointed.hasOwnProperty('current')){
513             this.pointed.current = this.pointed.items[this.position];
514         }
515         return returnvalue;
516     };
517     this.maximize = function(){
518         //Move the pointer to the end
519         this.started = true;
520         this.position = this.max-1;
521     };
522     this.minimize = function(){
523         //Move the pointer to the beginning
524         //this.started = true;
525         this.position = 0;
526     }
527 }
